00220 { \
00221   type_t *low, *high, *k, *ll, *hh, *mid; \
00222   low = arr; high = arr + n - 1; k = arr + kk; \
00223   for (;;) { \
00224       if (high <= low) return *k; \
00225       if (high == low + 1) { \
00226           if (__sort_lt(*high, *low)) TMAP_SORT_SWAP(type_t, *low, *high); \
00227           return *k; \
00228       } \
00229       mid = low + (high - low) / 2; \
00230       if (__sort_lt(*high, *mid)) TMAP_SORT_SWAP(type_t, *mid, *high); \
00231       if (__sort_lt(*high, *low)) TMAP_SORT_SWAP(type_t, *low, *high); \
00232       if (__sort_lt(*low, *mid)) TMAP_SORT_SWAP(type_t, *mid, *low); \
00233       TMAP_SORT_SWAP(type_t, *mid, *(low+1)); \
00234       ll = low + 1; hh = high; \
